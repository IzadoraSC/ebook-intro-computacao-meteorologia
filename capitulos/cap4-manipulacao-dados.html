  mutate(across(everything(), ~replace_na(.x, 0)))

#Substituir NAs por média da variável
dados_preenchidos2 <- dados_nc  |> 
  mutate(across(where(is.numeric), ~ifelse(is.na(.), mean(., na.rm = TRUE), .)))
